Package pyledlanglib

This module does simple thing it takes usual Python3 code and converts it to embedded PyLEDASM code. Developed by alex5250 just for fun.
Usage example:

from pyledlanglib import Compiler

def main():
        x=Compiler(output_file="main.bin",step=10,debug=False)
        for a in range(0,10):
                                x.set_color(a*25,a*25,a*25)
                                x.sleep(20)
        x.final_build()
pass
main()

"""
This module does simple thing it takes usual Python3 code and converts it to embedded PyLEDASM code.
Developed by alex5250 just for fun.  
Usage example:
```
from pyledlanglib import Compiler

def main():
        x=Compiler(output_file="main.bin",step=10,debug=False)
        for a in range(0,10):
                                x.set_color(a*25,a*25,a*25)
                                x.sleep(20)
        x.final_build()
pass
main()
```
"""
import sys,os
import platform
import time

class Compiler:
        """
    Compiler this is an main class that should be imported in case code needs to be generated.    \
        The Compiler class requires as parameters output_file path debug selection(true,false) and step as int.
    output_file: is an path where output file after compilation should been saved.
    debug: is an bool varible that sets should we include debug info or not
        step: is an varible that sets in milliseconds the delay been each sleeps. Depends on intrepeter realization.
    return: returns nothing
        """
        
        step:int=20 
        """default step setting for delay."""
        
        verison:str="v0.1.3" 
        """version constant"""
        debug:bool=True 
        """debug behavoir default constant"""
        time_run:int=0 
        """timer from begining of run"""


        def __init__(self,output_file:str,debug:bool,step:int):
                print("Welcome the compiler tool. ")
                print("=============> In progress <============")
                self.output_file=open(output_file,"wb")
                self.output_file_path=output_file
                print(f"Output file:{output_file}")
                self.debug=debug
                print("You are running PyLEDLangLib on top of Python 3")
                print(f"Python verison: {sys.version} PyLEDLangLib version: {self.verison} Platform: {platform.platform()}")
                print(f"Debug: {self.debug}")
                self.time_run=time.time()
                self.step=step
        pass
        
        def sleep(self,time:int):
                """
                
                The function that generates code in PyLEDASM for required amount of time delay. 
                As example if cycle is 20 and you need an delay 40ms as parameter (time as int) pass 40 and code will do convertion for you.
                """
                int_values=(int(time / self.step)) # convert to loop amount if need
                if(self.debug):
                        self.output_file.write(bytes(f';user defined delay block via sleep method: {time} cycles\n','UTF-8')) # write to file
                for a in range(0,int_values):
                        self.output_file.write(bytes(f'slp:0{a} -> sleep\n','UTF-8')) # write to file
        pass
        

        
        def final_build(self):
                """
                displays an finished build banner an finishes an record to file.
                good to include on end of your programm
                """
                print("=============> Build is successful <==============")
                print(f"Taken: {(time.time()-self.time_run)} seconds")
                print(f"Copy your {self.output_file_path} to SD and insert to controller")
                print("")


        def add_extra_byte(self,input_hex:str):
                """
                add extra byte is an function that is been used inside the code for 
                Simple function that adds zero if only one byte is returned
                check an len of input str and using string format add an zeros
                or return orignal value.
                """
                if(len(input_hex) < 2):
                        return f"0{input_hex}"
                else:
                        return input_hex
        pass

        def set_color(self,r:int,g:int,b:int):
                """
                converts an rgb to bin command.
                So at first get value from user then convert it to hex and remove 0x 
                prefix. Then convert it to upper case.
                using an add_extra_byte add an extra zero if needed. so it converts an
                an 0 => 00 then using string format make final string an save it to bin output.
                As param takes 3 ints in rgb from 0-255
                """
                if(self.debug):
                        self.output_file.write(bytes(f';user defined color from hex {r} {g} {b}\n','UTF-8')) # write to file
                red=str(hex(int(r))).replace("0x","").upper()
                green=str(hex(int(g))).replace("0x","").upper()
                blue=str(hex(int(b))).replace("0x","").upper()
                red=self.add_extra_byte(red)
                green=self.add_extra_byte(green)
                blue=self.add_extra_byte(blue)
                output_string=f"reg clr:{red}{green}{blue}\n"
                self.output_file.write(bytes(output_string,'UTF-8'))
        pass
        
        def set_hex_color(self,hex:str):
                """
                converts an hex to bin command.
                takes string of hex. converts it to upper case and prints to file
                """
                if(self.debug):
                        self.output_file.write(bytes(f';user defined color from hex {hex} \n','UTF-8')) # write to file
                output_string=f"reg clr:{hex.upper()}\n"
                self.output_file.write(bytes(output_string,'UTF-8'))
        pass
